364 return std::nullopt;

368 std::wistringstream ss(
369 std::wstring(static_cast<const wchar_t*>(verBuf), verLen));
370 std::wstring token;
371 for (std::size_t i = 0; i < 4 && std::getline(ss, token, L'.'); ++i) {
372 version[i] = static_cast<std::uint16_t>(std::stoi(token));
373 }

793 void unpack_file(binary_io::file_istream& a_in, header_t a_header)
794 {
795 std::uint8_t type = 0;
796 std::uint64_t id = 0;
797 std::uint64_t offset = 0;
798 std::uint64_t prevID = 0;
799 std::uint64_t prevOffset = 0;
800 for (auto& mapping : _id2offset) {
801 a_in.read(type);
802 const auto lo = static_cast<std::uint8_t>(type & 0xF);
803 const auto hi = static_cast<std::uint8_t>(type >> 4);
804
805 switch (lo) {
806 case 0:
807 a_in.read(id);
808 break;
809 case 1:
810 id = prevID + 1;
811 break;
812 case 2:
813 id = prevID + std::get<0>(a_in.read<std::uint8_t>());
814 break;
815 case 3:
816 id = prevID - std::get<0>(a_in.read<std::uint8_t>());
817 break;
818 case 4:
819 id = prevID + std::get<0>(a_in.read<std::uint16_t>());
820 break;
821 case 5:
822 id = prevID - std::get<0>(a_in.read<std::uint16_t>());
823 break;
824 case 6:
825 std::tie(id) = a_in.read<std::uint16_t>();
826 break;
827 case 7:
828 std::tie(id) = a_in.read<std::uint32_t>();
829 break;
830 default:
831 stl::report_and_fail("unhandled type"sv);
832 }
833
834 const std::uint64_t tmp = (hi & 8) != 0 ? (prevOffset / a_header.pointer_size()) : prevOffset;
835
836 switch (hi & 7) {
837 case 0:
838 a_in.read(offset);
839 break;
840 case 1:
841 offset = tmp + 1;
842 break;
843 case 2:
844 offset = tmp + std::get<0>(a_in.read<std::uint8_t>());
845 break;
846 case 3:
847 offset = tmp - std::get<0>(a_in.read<std::uint8_t>());
848 break;
849 case 4:
850 offset = tmp + std::get<0>(a_in.read<std::uint16_t>());
851 break;
852 case 5:
853 offset = tmp - std::get<0>(a_in.read<std::uint16_t>());
854 break;
855 case 6:
856 std::tie(offset) = a_in.read<std::uint16_t>();
857 break;
858 case 7:
859 std::tie(offset) = a_in.read<std::uint32_t>();
860 break;
861 default:
862 stl::report_and_fail("unhandled type"sv);
863 }
864
865 if ((hi & 8) != 0) {
866 offset *= a_header.pointer_size();
867 }
868
869 mapping = { id, offset };
870
871 prevOffset = offset;
872 prevID = id;
873 }
874 }
